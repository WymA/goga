\begin{Algo} [t] \ALGOSIZE
    \DontPrintSemicolon
    \KwIn{trial solutions $A$ and $B$}
    \KwOut{returns {\bf true} if $A$ wins}

    \BlankLine
    \Lcom{compare solutions using OOR and OVA}
    $A_{dom}, B_{dom} \gets \text{\FnCompareSolutions}(A, B)$\;
    \lIf{$A_{dom}$}{ \KwRet true }
    \lIf{$B_{dom}$}{ \KwRet false }

    \BlankLine
    \Lcom{tie: single-OVA}
    \If{$N_f=1$}{
        \lIf{$\eta_A > \eta_B$}{ \KwRet true }
        \lIf{$\eta_B > \eta_A$}{ \KwRet false }
		\KwRet~\FnFlipCoin(0.5)\;
	}

    \BlankLine
    \Lcom{tie: multi-OVA: same Pareto front}
    \If{$\phi_A = \phi_B$}{
		\lIf{$w_A > w_B$}{ \KwRet true }
		\lIf{$w_B > w_A$}{ \KwRet false }
		\KwRet~\FnFlipCoin(0.5)\;
	}

    \BlankLine
    \Lcom{tie: multi-OVA: different fronts}
    \lIf{$\phi_A < \phi_B$}{ \KwRet true }
    \lIf{$\phi_B < \phi_A$}{ \KwRet false }
    \lIf{$\eta_A > \eta_B$}{ \KwRet true }
    \lIf{$\eta_B > \eta_A$}{ \KwRet false }

    \BlankLine
    \Lcom{tie: nothing else can be done}
	\KwRet~\FnFlipCoin(0.5)\;

\caption{
    \FnFight($A,B$)
    performs the comparison between trial solutions for a tournament. All data such as OORs, OVAs
    and measures of diversity are used.
}
\label{alg:fight}
\end{Algo}
