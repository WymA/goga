\begin{Algo} [t] \ALGOSIZE
    \DontPrintSemicolon
    \KwIn{trial solutions $A,B,a,b$}
    \KwOut{$\Omega$: replace $A$ or $B$, or both, in the right position in the set $\Omega$}

    compute distances $d_{Aa}$, $d_{Ab}$, $d_{Ba}$ and $d_{Bb}$\;
	\eIf{$d_{Aa}+d_{Bb} < d_{Ab}+d_{Ba}$}{
        \If(\LcomIf{$a$ wins over $A$}){\KwNot $\text{\FnFight}(A,a)$}{
            replace $A$ with $a$ in $\Omega$\;
        }
        \If(\LcomIf{$b$ wins over $B$}){\KwNot $\text{\FnFight}(B,b)$}{
            replace $B$ with $b$ in $\Omega$\;
        }
    }{
        \If(\LcomIf{$b$ wins over $A$}){\KwNot $\text{\FnFight}(A,b)$}{
            replace $A$ with $b$ in $\Omega$\;
        }
        \If(\LcomIf{$a$ wins over $B$}){\KwNot $\text{\FnFight}(B,a)$}{
            replace $B$ with $a$ in $\Omega$\;
        }
    }

\caption{
    \FnTournament($\Omega$, \, $A,B,a,b$)
    performs the tournament with replacement of winners in $\Omega$.
}
\label{alg:tournament}
\end{Algo}
