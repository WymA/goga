\begin{Algo} [t] \ALGOSIZE
    \DontPrintSemicolon
    \KwIn{set of trial solutions $\Omega$}
    \KwOut{set of solutions in Pareto fronts $F_{i,j}$, size of fronts $z_i$, and ranks $\phi_i$}
    zero all $N_{wins}^i$, $N_{loss}^i$, $\phi_i$ and $z_i$ variables\;

    \BlankLine
    \Lcom{compute wins/losses data}
    \For{each solution $A$ in $\Omega$}{
        \For{each $B$ in $\Omega$ with index greater than $A$}{
            $A_{dom}, B_{dom} \gets \text{\FnCompareSolutions}(A, B)$\;
            \If{$A_{dom}$}{
                $W_A[N_{wins}^A] \gets B$\;
                $N_{wins}^A \pluseq 1$\;
                $N_{loss}^B \pluseq 1$\;
            }
            \If{$B_{dom}$}{
                $W_B[N_{wins}^B] \gets A$\;
                $N_{wins}^B \pluseq 1$\;
                $N_{loss}^A \pluseq 1$\;
            }
        }
    }

    \BlankLine
    \Lcom{first front}
    \For{each solution $A$ in $\Omega$}{
        \If{$N_{loss}^A = 0$}{
            $F_{0,{z_0}} \gets A$\;
            $z_0 \pluseq 1$\;
        }
    }

    \BlankLine
    \Lcom{next fronts}
    \For{each front index $0 \leq r < N_{sol}$}{
        \If{$z_r=0$}{
            {\bf break}\;
        }
        $s \gets r + 1$\;
        \For{solution indices $0 \leq i < z_r$}{
            $A \gets F_{r,i}$\;
            \For{each solution $B$ in $W_A$}{
                $N_{loss}^B \minuseq 1$\;
                \If(\LcomIf{$B$ in next F}){$N_{loss}^B = 0$}{
                    $\phi_B \gets s$\;
                    $F_{s,{z_s}} \gets B$\;
                    $z_s \pluseq 1$\;
                }
            }
        }
    }
\caption{
    \FnParetoFronts($\Omega$)
    computes Pareto fronts for a set of trial solutions $\Omega$.
}
\label{alg:fronts}
\end{Algo}
