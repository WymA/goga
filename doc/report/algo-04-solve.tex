\begin{Algo} [t] \ALGOSIZE
    \DontPrintSemicolon
    \KwIn{configuration parameters}
    \KwOut{optimal set of solutions $S$}
    $t \gets 0$ \Lcom{evolution pseudo time}
    $t_{exc} \gets \Delta t_{exc}$ \Lcom{exchange time}
    \While{$t < t_{max}$}{

        \Lcom{parallel execution up $t_{exc}$}
        \For{each group $i$}{
            \Begin{
                \For{$time \gets t$ \KwTo $time<t_{exc}$} {
                    \FnEvolveOneGroup($I$)\;
                }
                send \emph{done} flag through \emph{channel}\;
            }
        }
        \For{each group}{
            collect \emph{done} flag from \emph{channel}\;
        }

        \BlankLine
        \Lcom{compute metrics}
        \FnMetrics($S$)\;

        \BlankLine
        \Lcom{exchange solutions via tournament}
        \For{each group $i$}{
            $j \gets (i+1) \,\%\, N_{cpu}$\;
            $A,B     \gets $ random solutions from $G_i$\;
            $a,\;b\; \gets $ random solutions from $G_j$\;
            \FnTournament($S$, \, $A,B,a,b$)\;
        }

        \BlankLine
        \Lcom{exchange one solution randomly}
        select non-repeated pairs of groups\;
        \For{each random pair $(G_0,G_1)$}{
            $A \gets $ random solution from $G_0$\;
            $B \gets $ random solution from $G_1$\;
            swap $A$ by $B$ in $S$\;
        }

        \BlankLine
        \Lcom{update time variables}
        $t \pluseq \Delta t_{exc}$\;
        $t_{exc} \pluseq \Delta t_{exc}$\;
    }
\caption{
    \FnSolve()
    solves the optimisation problem.
}
\label{alg:main}
\end{Algo}
